Export to Online Repository

action: /api2/v1/connectors/{connectorId}/folders/{encodedName}
                          
                           // method POST needs to be used
                           // multipart/form-data is required as a Content-Type
                           // boundary needs to be correctly set in both header and body to process the request

sample body

--uBPJE!%vdZ*leTfePa%Rro&6
Content-Disposition: form-data; name="file"; filename="testfile2.txt"
Content-Type: application/octet-stream

<title>A post with "quotes"</title>Something.
--uBPJE!%vdZ*leTfePa%Rro&6--

response: empty

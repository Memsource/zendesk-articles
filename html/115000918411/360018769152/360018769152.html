Create a New Quote

action: api/v2/quote/create
    token                       string
    name                        string
    project                     domain(Project)
    analyse                     domain(Analyse)
    priceList                   domain(PriceList)
    discountScheme              domain(DiscountScheme)      O
    workflowStep                list(domain(WorkflowStep))  O
                        

response: JSON
    {
      "id": 238                 // id of the new quote
    }

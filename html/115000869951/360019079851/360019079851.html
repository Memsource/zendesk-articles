Check

action: api/v2/spellCheck/check
    token                    string
    lang                     lang
    text                     list(string)
    referenceText            list(string)                     O
    zeroLengthSeparator      string                           O

response: JSON
   [
     {
       "text":"hello",
       "misspelledWords":[]
     },
     {
       "text":"World",
       "misspelledWords":[]
     },
     {
       "text":"foox hella",
       "misspelledWords":[
         {
           "word":"foox",
           "offset":0
         },
         {
           "word":"hella",
           "offset":5
         }
       ]
     }
   ]

error codes
    LanguageNotSupported
